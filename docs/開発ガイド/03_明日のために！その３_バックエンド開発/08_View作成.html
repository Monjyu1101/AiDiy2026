08_View作成

V系は DBの CREATE VIEW ではなく、Routerで生SQLを実行して一覧を返す方針です。

方針

  • V系は表示用途の読み取り専用APIとして実装する
  • sqlalchemy.text() で明示的にSQLを実行する
  • 戻り値は items/total/limit を含む統一形式
  • リクエストには ListRequest（件数制限・無効も表示）を受け取る
  • 有効フラグのフィルタとLIMIT句は list_controls.py のユーティリティ関数で統一管理する

list_controls.py ユーティリティ

backend_server/list_controls.py は一覧系APIで共通利用するヘルパーモジュールです。

# backend_server/list_controls.py（抜粋）
LIMITED_MAX_ITEMS = 1000

def is_limit_enabled(request) -> bool:
    """リクエストの件数制限フラグを読み取る（デフォルト: True）"""
    return _to_bool(getattr(request, "件数制限", None), True)

def should_show_inactive(request) -> bool:
    """無効も表示フラグを読み取る（デフォルト: False）"""
    return _to_bool(getattr(request, "無効も表示", None), False)

def get_list_limit(request) -> Optional[int]:
    """LIMIT値を返す（件数制限=Trueなら1000、Falseならなし）"""
    if is_limit_enabled(request):
        return LIMITED_MAX_ITEMS
    return None

def get_limit_clause(request) -> tuple[str, Optional[int]]:
    """SQL用のLIMIT句文字列とバインド値のタプルを返す"""
    limit_value = get_list_limit(request)
    if limit_value is None:
        return "", None
    return "LIMIT :limit", limit_value

def append_active_condition(conditions: list[str], request, column_name: str) -> None:
    """無効も表示=Falseのとき 'カラム名 = 1' をWHERE条件リストに追加する"""
    if not should_show_inactive(request):
        conditions.append(f"{column_name} = 1")

def apply_active_filter(query, model, request):
    """ORM queryに有効=1フィルタを追加する（M系用）"""
    if should_show_inactive(request):
        return query
    return query.filter(model.有効 == True)

V系一覧エンドポイントの雛形（list_controls.py 利用版）

以下が最新の実装パターンです。list_controls をインポートして件数制限・有効フィルタを一元管理します。

from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from sqlalchemy import text
from typing import Optional
import apps_schema as schemas, deps, apps_models as models
from list_controls import append_active_condition, get_limit_clause

router = APIRouter(prefix="/apps/V商品", tags=["V商品"])

@router.post("/一覧", response_model=schemas.ResponseBase)
def list_V商品(
    request: Optional[schemas.ListRequest] = None,
    db: Session = Depends(deps.get_db),
    現在利用者: models.C利用者 = Depends(deps.get_現在利用者)
):
    conditions = []
    # 有効フラグフィルタ（request.無効も表示=Falseの場合に 有効=1 条件を追加）
    append_active_condition(conditions, request, 's."有効"')
    where_sql = f"WHERE {' AND '.join(conditions)}" if conditions else ""

    # LIMIT句（request.件数制限=Trueの場合に LIMIT 1000 を追加）
    limit_sql, limit_value = get_limit_clause(request)
    params = {}
    if limit_value is not None:
        params["limit"] = limit_value

    sql = f"""
    SELECT s.*, m."商品分類名"
    FROM "M商品" s
    LEFT JOIN "M商品分類" m ON s."商品分類ID" = m."商品分類ID"
    {where_sql}
    ORDER BY s."商品ID"
    {limit_sql}
    """
    result = db.execute(text(sql), params).fetchall()

    count_sql = f"""
    SELECT count(*) FROM "M商品" s
    LEFT JOIN "M商品分類" m ON s."商品分類ID" = m."商品分類ID"
    {where_sql}
    """
    total = db.execute(text(count_sql), params).scalar()

    return schemas.ResponseBase(
        status="OK",
        message="V商品一覧を取得しました",
        data={"items": [dict(row._mapping) for row in result], "total": total, "limit": limit_value}
    )

ポイント:

  • request: Optional[schemas.ListRequest] = None — リクエストなし（省略可）でもデフォルト動作（件数制限あり・有効のみ）が適用される
  • append_active_condition の第3引数はSQL上のエイリアス付きカラム名（例: s."有効"）で指定する
  • 件数SQLの params は WHERE条件のバインド変数と共用しているため、count側でも同じ params を渡す
  • 行データは dict(row._mapping) で辞書変換する（古い row.カラム名 参照より安全）

  • 件数SQL（count）もセットで管理する
